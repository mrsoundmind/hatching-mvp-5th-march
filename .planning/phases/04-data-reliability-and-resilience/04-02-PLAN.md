**server/storage.ts — Update `getMessagesByConversation` in BOTH MemStorage and DatabaseStorage:**

The key change: when `limit` is provided WITHOUT `page`, return the LAST `limit` messages (most recent), sorted ascending. When `before` is provided, return the last `limit` messages BEFORE that cursor timestamp, sorted ascending.

For **MemStorage** (~line 811):
```typescript
async getMessagesByConversation(conversationId: string, options?: {
  page?: number; limit?: number; before?: string; after?: string; messageType?: string;
}): Promise<Message[]> {
  let msgs = Array.from(this.messages.values())
    .filter(m => m.conversationId === conversationId);

  if (options?.messageType) msgs = msgs.filter(m => m.messageType === options.messageType);

  // Sort ascending by createdAt
  msgs.sort((a, b) => new Date(a.createdAt).getTime() - new Date(b.createdAt).getTime());

  // Cursor-based: filter messages before the cursor timestamp
  if (options?.before) {
    const cutoff = new Date(options.before).getTime();
    msgs = msgs.filter(m => new Date(m.createdAt).getTime() < cutoff);
  }
  if (options?.after) {
    const cutoff = new Date(options.after).getTime();
    msgs = msgs.filter(m => new Date(m.createdAt).getTime() > cutoff);
  }

  // If limit is set (and no explicit page), return the LAST N messages (most recent batch)
  const limit = options?.limit || msgs.length;
  if (options?.page) {
    const start = (options.page - 1) * limit;
    msgs = msgs.slice(start, start + limit);
  } else {
    // Return last `limit` items (most recent window), preserving ascending sort
    msgs = msgs.slice(Math.max(0, msgs.length - limit));
  }

  return msgs;
}
```

For **DatabaseStorage** (~line 1538): Apply the same logic. Currently it fetches ALL rows then filters/sorts/slices in JS. Replace with the same cursor-aware logic (still fetching all rows from DB for now — SQL-level optimization is a future improvement, but the slicing logic must be correct):
```typescript
async getMessagesByConversation(conversationId: string, options?: {
  page?: number; limit?: number; before?: string; after?: string; messageType?: string;
}): Promise<Message[]> {
  const rows = await db.select().from(schema.messages)
    .where(eq(schema.messages.conversationId, conversationId));
  let msgs = rows as Message[];

  if (options?.messageType) msgs = msgs.filter(m => m.messageType === options.messageType);

  msgs.sort((a, b) => new Date(a.createdAt).getTime() - new Date(b.createdAt).getTime());

  if (options?.before) {
    const cutoff = new Date(options.before).getTime();
    msgs = msgs.filter(m => new Date(m.createdAt).getTime() < cutoff);
  }
  if (options?.after) {
    const cutoff = new Date(options.after).getTime();
    msgs = msgs.filter(m => new Date(m.createdAt).getTime() > cutoff);
  }

  const limit = options?.limit || msgs.length;
  if (options?.page) {
    const start = (options.page - 1) * limit;
    msgs = msgs.slice(start, start + limit);
  } else {
    msgs = msgs.slice(Math.max(0, msgs.length - limit));
  }

  return msgs;
}
```
